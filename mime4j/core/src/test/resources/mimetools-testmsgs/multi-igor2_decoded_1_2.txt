install:
	@if [ ! -d ${SITE_PERL} ]; then \
	    echo "Please edit the SITE_PERL in your Makefile"; exit -1; \
        fi          
	@if [ ! -w ${SITE_PERL} ]; then \
	    echo "No permission... should you be root?"; exit -1; \
        fi          
	@if [ ! -d ${SITE_PERL}/MIME ]; then \
	    mkdir ${SITE_PERL}/MIME; \
        fi
	install -m 0644 MIME/*.pm ${SITE_PERL}/MIME
